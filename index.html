    package ca.ciccc;

    import java.util.Scanner; //utilitary class to access the keyboard

    public class TemperatureConv {

      public static void main(String[] args) {

        System.out.println("=========================================");
        System.out.println("|    TEMPERATURE CONVERSION             |");
        System.out.println("=========================================");

        int op = -1;
        Scanner input = new Scanner(System.in);

        while (op != 0) {
          System.out.println("Choose one option:");
          System.out.println("1 - Celcius to Fahrenheit");
          System.out.println("2 - Fahrenheit to Celcius");
          System.out.println("0 - to exit");
          System.out.println("\n");

          op = input.nextInt();

          if (op == 1) {
            // Celcius to Farenheit
            System.out.println("Ok, Celcius to Fahrenheit...");

            System.out.println("Please, what is the temperature in Celcius?");
            int c = input.nextInt();

            if (c < 0 || c > 100) {
              System.out.println("Try another number");
              continue;
            }
            // calculate fahrenheit from celcius
            double f = (c * 9 / 5) + 32;

            System.out.println(c + " in Celcius is " + f + " in Fahrenheit.");
            System.out.println("Next...");

          } else if (op == 2) {
            // Fharenheit to Celcius
            System.out.println("Ok, Fahrenheit to Celcius...");

            System.out.println("Please, what is the temperature in Fahrenheit?");
            int f = input.nextInt();

            if (f < 32 || f > 212) {
              System.out.println("Try another number");
              continue;
            }
            // calculate celcius from fahrenheit
            double c = (f - 32) * 5 / 9;

            System.out.println(f + " in Fahrenheit is " + c + " in Celcius.");
            System.out.println("Next...");
          }
        } // end of while loop
        input.close(); // closing the scanner input
        System.out.println("=========================================");
        System.out.println("|    Thank you, bye!!!                  |");
        System.out.println("=========================================");
      }
    }
